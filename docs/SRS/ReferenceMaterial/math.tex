\subsection{Mathematical Notation}

\begin{center}

    \renewcommand{\arraystretch}{1.2}
    \begin{tabular}{C{0.2\textwidth} m{0.7\textwidth}}
        \toprule
        \textbf{Notation} & \textbf{Description}\\
        \midrule

        \rowcolor[gray]{0.9}$\neg x$ &  NOT x \\

        $x \land y$ & x AND y \\

        \rowcolor[gray]{0.9}$x \lor y$ & x OR y \\

        $\forall x$ & Universal quantifier \\

        \rowcolor[gray]{0.9}$\exists x$ & Existential quantifier \\

        $\left[x,y\right]$ & A closed interval bounded by $x$ and $y$,
        inclusive \\

        \rowcolor[gray]{0.9}${\left(x,y\right)}$ & An open interval bounded by
        $x$ and $y$, exclusive \\

        $|x|$ & Absolute value of $x$ \\

        \rowcolor[gray]{0.9}$\mathlarger\sum_{i=x}^{y} s_i$ & Summation of
        $s_i$ from $i = x$ to $y$ inclusive \\

        $x \cdot y$ & Multiply $x$ and $y$ \\

        \rowcolor[gray]{0.9}$P(B|A)$ & Conditional probability of $B$ given $A$
        \\

        $\left\{x_0, x_1, ..., x_n\right\}$ & A set of $n$ elements \\

        \rowcolor[gray]{0.9}$\left\{X\right\}$ & A set of elements of type $X$
        \\

        $X \subset Y $ & $X$ is a proper/strict subset of $Y$\\

        \rowcolor[gray]{0.9}$X \subseteq Y $ & $X$ is a subset of $Y$ \\

        $A \times B$ & The Cartesian product of $A$ and $B$ \\

        \rowcolor[gray]{0.9}$\left(x_0, x_1, ..., x_n\right)$ & A sequence of
        $n$ elements \\

        $x : X$ & A variable $x$ of type $X$ \\

        \rowcolor[gray]{0.9}$x : X \rightarrow y : Y$ & A function mapping $x :
        X$ to $y : Y$ \\

        $x \oplus y$ & Shorthand for $\mathtt{apply}(x,
        y)$, where $\mathtt{apply}$ is a function that changes $x$ by $y$ \\

        \rowcolor[gray]{0.9}$X^?$ & An Option/Maybe type that returns a value
        of type $X$ or Nothing/None \\

        $\left< e_1, ..., e_n \right>$ & An enumeration with $n$ labels \\

        \rowcolor[gray]{0.9}$\left\{ l_1 = v_1, ..., l_n = v_n \right\}$ & A
        record with $n$ elements, each with a label $l$ and value $v$ such that
        no two sets of $l$ and $v$ must have the same type \\

        $\left\{ r \text{ with } x = y \right\}$ & Update a record $r$ that has
        the label $x$ with the value $y$ \\

        \rowcolor[gray]{0.9}$X \defEq y $ & Assignment of $y$ to $X$; $X$
        ``is defined by'' $y$\\

        \bottomrule
    \end{tabular}

\end{center}
